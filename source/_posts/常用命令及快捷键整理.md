python快排:
def quick_sort(lst):
    if len(lst) <= 1:
        return lst
    else:
        pivot = lst[0]
        return quick_sort([x for x in lst[1:] if x < pivot]) + [pivot] + quick_sort([y for y in lst[1:] if y >= pivot])
